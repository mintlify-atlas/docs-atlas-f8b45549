## Overview

The Social API powers Plotlist's social features including follower relationships, likes, comments, activity feeds, and contact-based user discovery.

### Mutations

#### follow

Follow a user.

```typescript
export const follow = mutation({
  args: { userIdToFollow: v.id("users") },
  handler: async (ctx, args) => { ... },
});
```

<ParamField path="userIdToFollow" type="Id<'users'>" required>
  User to follow
</ParamField>

**Returns:** Follow ID.

**Rate limiting:** 30 follows per minute.

**Side effects:**
- Increments follower's `countsFollowing`
- Increments followee's `countsFollowers`
- Idempotent (returns existing follow if already following)

#### unfollow

Unfollow a user.

```typescript
export const unfollow = mutation({
  args: { userIdToUnfollow: v.id("users") },
  handler: async (ctx, args) => { ... },
});
```

**Side effects:**
- Decrements follower's `countsFollowing`
- Decrements followee's `countsFollowers`

## Contact Sync

### Actions

#### syncSnapshot

Sync phone contacts to find friends on Plotlist.

```typescript
export const syncSnapshot = action({
  args: {
    entries: v.array(
      v.object({
        sourceRecordId: v.optional(v.string()),
        displayName: v.optional(v.string()),
        phone: v.string(),
      }),
    ),
  },
  handler: async (ctx, args) => { ... },
});
```

<ParamField path="entries" type="array" required>
  Contact entries with phone numbers
</ParamField>

**Returns:**

<ResponseField name="totalCount" type="number" required>
  Total contacts synced
</ResponseField>

<ResponseField name="matchedCount" type="number" required>
  Contacts matched to users
</ResponseField>

<ResponseField name="inviteCount" type="number" required>
  Contacts not on Plotlist
</ResponseField>

<ResponseField name="syncedAt" type="number" required>
  Sync timestamp
</ResponseField>

**Rate limiting:** 6 syncs per 10 minutes.

**Limits:** Maximum 2000 contacts per sync.

**Privacy:**
- Phone numbers are hashed before storage
- Only matched user IDs are stored, not phone numbers
- Contact data is deleted on account deletion

## Example Usage

### Follow User

```typescript
import { useMutation, useQuery } from "convex/react";
import { api } from "../convex/_generated/api";

function FollowButton({ userId }: { userId: Id<"users"> }) {
  const isFollowing = useQuery(api.follows.isFollowing, { userId });
  const follow = useMutation(api.follows.follow);
  const unfollow = useMutation(api.follows.unfollow);
  
  const handleClick = async () => {
    if (isFollowing) {
      await unfollow({ userIdToUnfollow: userId });
    } else {
      await follow({ userIdToFollow: userId });
    }
  };
  
  return (
    <button onClick={handleClick}>
      {isFollowing ? "Unfollow" : "Follow"}
    </button>
  );
}
```

### Like/Unlike

```typescript
import { useMutation, useQuery } from "convex/react";
import { api } from "../convex/_generated/api";

function LikeButton({ 
  reviewId 
}: { 
  reviewId: Id<"reviews"> 
}) {
  const liked = useQuery(api.likes.getForUserTarget, {
    targetType: "review",
    targetId: reviewId,
  });
  const toggle = useMutation(api.likes.toggle);
  
  const handleClick = async () => {
    await toggle({
      targetType: "review",
      targetId: reviewId,
    });
  };
  
  return (
    <button onClick={handleClick}>
      {liked ? "❤️" : "🤍"}
    </button>
  );
}
```

### Contact Sync

```typescript
import { useAction, useQuery } from "convex/react";
import { api } from "../convex/_generated/api";

function ContactSync() {
  const status = useQuery(api.contacts.getStatus);
  const sync = useAction(api.contacts.syncSnapshot);
  const matches = useQuery(api.contacts.getMatches, { limit: 20 });
  
  const handleSync = async () => {
    // Get contacts from device
    const contacts = await getDeviceContacts();
    
    const result = await sync({
      entries: contacts.map(c => ({
        phone: c.phoneNumber,
        displayName: c.name,
      })),
    });
    
    console.log(`Found ${result.matchedCount} friends!`);
  };
  
  return (
    <div>
      <button onClick={handleSync}>Sync Contacts</button>
      {status?.hasSynced && (
        <div>
          <p>{status.matchedCount} friends on Plotlist</p>
          <UserList users={matches ?? []} />
        </div>
      )}
    </div>
  );
}
```
