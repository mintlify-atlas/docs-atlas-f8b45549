## Overview

The Lists API manages user-created collections of TV shows. Lists can be public or private, support custom ordering, and include optional cover images and descriptions.

## Tables

### lists

List metadata and settings.

<ResponseField name="ownerId" type="Id<'users'>" required>
  User who created the list
</ResponseField>

<ResponseField name="title" type="string" required>
  List title (max 120 characters)
</ResponseField>

<ResponseField name="description" type="string">
  List description (max 2000 characters)
</ResponseField>

<ResponseField name="isPublic" type="boolean" required>
  Whether list is publicly visible
</ResponseField>

<ResponseField name="coverStorageId" type="Id<'_storage'>">
  Optional cover image
</ResponseField>

### listItems

Shows within a list, with ordering.

<ResponseField name="listId" type="Id<'lists'>" required>
  Parent list
</ResponseField>

<ResponseField name="showId" type="Id<'shows'>" required>
  Show in the list
</ResponseField>

<ResponseField name="position" type="number" required>
  Sort order (1-based, lower = earlier)
</ResponseField>

<ResponseField name="addedAt" type="number" required>
  Timestamp when show was added
</ResponseField>

## List Queries

### get

Get a list by ID.

```typescript
export const get = query({
  args: { listId: v.id("lists") },
  handler: async (ctx, args) => { ... },
});
```

<ParamField path="listId" type="Id<'lists'>" required>
  List ID
</ParamField>

**Returns:** List document or `null`.

**Errors:**
- `"Not allowed"` - Private list and viewer is not owner

### listForUser

Get authenticated user's lists.

```typescript
export const listForUser = query({
  args: { 
    userId: v.id("users"), 
    paginationOpts: paginationOptsValidator 
  },
  handler: async (ctx, args) => { ... },
});
```

<ParamField path="userId" type="Id<'users'>" required>
  Must be authenticated user's ID
</ParamField>

**Returns:** Paginated lists ordered by `updatedAt` (most recent first).

**Errors:**
- `"Not allowed"` - userId doesn't match authenticated user

### listPublicForUser

Get public lists by a user.

```typescript
export const listPublicForUser = query({
  args: { 
    userId: v.id("users"), 
    paginationOpts: paginationOptsValidator 
  },
  handler: async (ctx, args) => { ... },
});
```

**Returns:** Paginated public lists only.

```typescript
  },
  handler: async (ctx, args) => { ... },
```

<ParamField path="listId" type="Id<'lists'>" required>

</ParamField>

</ParamField>

**Rate limiting:** 60 additions per minute.

**Errors:**
- `"Not allowed"` - User doesn't own this list

## Example Usage

### Create a List

```typescript
import { useMutation } from "convex/react";
import { api } from "../convex/_generated/api";

function CreateListForm() {
  const createList = useMutation(api.lists.create);
  
  const handleSubmit = async (data: {
    title: string;
    description?: string;
    isPublic: boolean;
  }) => {
    const listId = await createList(data);
    console.log("Created list:", listId);
  };
  
  return <Form onSubmit={handleSubmit} />;
}
```

### Display List Items

```typescript
import { useQuery } from "convex/react";
import { api } from "../convex/_generated/api";

function ListView({ listId }: { listId: Id<"lists"> }) {
  const list = useQuery(api.lists.get, { listId });
  const items = useQuery(api.listItems.listDetailed, { listId });
  
  if (!list || !items) return <Loading />;
  
  return (
    <div>
      <h1>{list.title}</h1>
      {list.description && <p>{list.description}</p>}
      <div className="items">
        {items.map(({ item, show }) => (
          <ShowCard key={item._id} show={show} />
        ))}
      </div>
    </div>
  );
}
```

### Reorder List Items

```typescript
import { useMutation } from "convex/react";
import { api } from "../convex/_generated/api";
import { DragDropContext, Droppable, Draggable } from "react-beautiful-dnd";

function ReorderableList({ listId, items }: { 
  listId: Id<"lists">; 
  items: ListItem[];
}) {
  const reorder = useMutation(api.listItems.reorder);
  
  const handleDragEnd = async (result: any) => {
    if (!result.destination) return;
    
    const reordered = Array.from(items);
    const [removed] = reordered.splice(result.source.index, 1);
    reordered.splice(result.destination.index, 0, removed);
    
    await reorder({
      listId,
      orderedItemIds: reordered.map(item => item._id),
    });
  };
  
  return (
    <DragDropContext onDragEnd={handleDragEnd}>
      <Droppable droppableId="list">
        {(provided) => (
          <div ref={provided.innerRef} {...provided.droppableProps}>
            {items.map((item, index) => (
              <Draggable key={item._id} draggableId={item._id} index={index}>
                {(provided) => (
                  <div
                    ref={provided.innerRef}
                    {...provided.draggableProps}
                    {...provided.dragHandleProps}
                  >
                    <ListItem item={item} />
                  </div>
                )}
              </Draggable>
            ))}
            {provided.placeholder}
          </div>
        )}
      </Droppable>
    </DragDropContext>
  );
}
```

### Check List Membership

```typescript
import { useQuery } from "convex/react";
import { api } from "../convex/_generated/api";

function ListMembershipBadge({ showId }: { showId: Id<"shows"> }) {
  const memberOf = useQuery(api.listItems.getShowMembership, { showId });
  
  if (!memberOf || memberOf.length === 0) return null;
  
  return (
    <div className="badge">
      In {memberOf.length} {memberOf.length === 1 ? "list" : "lists"}
    </div>
  );
}
```
