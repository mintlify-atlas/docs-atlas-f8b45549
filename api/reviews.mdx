## Overview

The Reviews API manages user reviews and ratings for both entire shows and individual episodes. Reviews support ratings from 0.5 to 5.0 stars, optional text, spoiler warnings, and integration with the social feed.

## Rating Scale

Ratings use a 0.5 to 5.0 scale in 0.5 increments:
- 0.5 = Terrible
- 1.0-2.0 = Poor to Below Average
- 2.5-3.0 = Average to Good
- 3.5-4.5 = Very Good to Excellent
- 5.0 = Masterpiece

## Mutations

### create

Create a new show review.

```typescript
export const create = mutation({
  args: {
    showId: v.id("shows"),
    rating: v.number(),
    reviewText: v.optional(v.string()),
    spoiler: v.boolean(),
    seasonNumber: v.optional(v.number()),
    episodeNumber: v.optional(v.number()),
    episodeTitle: v.optional(v.string()),
  },
  handler: async (ctx, args) => { ... },
});
```

<ParamField path="showId" type="Id<'shows'>" required>
  Show being reviewed
</ParamField>

<ParamField path="rating" type="number" required>
  Rating from 0.5 to 5.0 (in 0.5 increments)
</ParamField>

<ParamField path="reviewText" type="string">
  Written review (max 5000 characters)
</ParamField>

<ParamField path="spoiler" type="boolean" required>
  Whether review contains spoilers
</ParamField>

<ParamField path="seasonNumber" type="number">
  Season number for episode reviews
</ParamField>

<ParamField path="episodeNumber" type="number">
  Episode number for episode reviews
</ParamField>

<ParamField path="episodeTitle" type="string">
  Episode title for display
</ParamField>

**Returns:** Review ID.

**Constraints:**
- One review per user per show (for show-level reviews)
- One review per user per episode (for episode reviews)
- Rating must be between 0.5 and 5.0

**Rate limiting:** 10 reviews per hour.

**Side effects:**
- Increments user's `countsReviews`
- Creates feed items for followers (show-level reviews only)
- Clears user taste caches/profiles

**Errors:**
- `"Rating must be between 0.5 and 5"`
- `"Review already exists for this show"`
- `"Review already exists for this episode"`

## Example Usage

### Create a Show Review

```typescript
import { useMutation } from "convex/react";
import { api } from "../convex/_generated/api";

function ReviewForm({ showId }: { showId: Id<"shows"> }) {
  const createReview = useMutation(api.reviews.create);
  
  const handleSubmit = async (data: {
    rating: number;
    text: string;
    hasSpoilers: boolean;
  }) => {
    await createReview({
      showId,
      rating: data.rating,
      reviewText: data.text,
      spoiler: data.hasSpoilers,
    });
  };
  
  return <Form onSubmit={handleSubmit} />;
}
```

### Display Show Reviews

```typescript
import { useQuery } from "convex/react";
import { api } from "../convex/_generated/api";

function ShowReviews({ showId }: { showId: Id<"shows"> }) {
  const reviews = useQuery(api.reviews.listForShowDetailed, {
    showId,
    paginationOpts: { numItems: 10, cursor: null },
  });
  
  if (!reviews) return <Loading />;
  
  return (
    <div>
      {reviews.page.map(({ review, author, authorAvatarUrl }) => (
        <ReviewCard
          key={review._id}
          review={review}
          author={author}
          avatarUrl={authorAvatarUrl}
        />
      ))}
    </div>
  );
}
```

### Rate an Episode

```typescript
import { useMutation } from "convex/react";
import { api } from "../convex/_generated/api";

function EpisodeRating({ 
  showId, 
  season, 
  episode 
}: { 
  showId: Id<"shows">; 
  season: number; 
  episode: number; 
}) {
  const rateEpisode = useMutation(api.reviews.rateEpisode);
  
  const handleRate = async (rating: number) => {
    await rateEpisode({
      showId,
      seasonNumber: season,
      episodeNumber: episode,
      rating,
    });
  };
  
  return <StarRating onChange={handleRate} />;
}
```

### Show Episode Stats

```typescript
import { useQuery } from "convex/react";
import { api } from "../convex/_generated/api";

function EpisodeStats({ 
  showId, 
  season, 
  episode 
}: { 
  showId: Id<"shows">; 
  season: number; 
  episode: number; 
}) {
  const stats = useQuery(api.reviews.getEpisodeStats, {
    showId,
    seasonNumber: season,
    episodeNumber: episode,
  });
  
  if (!stats) return null;
  
  return (
    <div>
      <StarDisplay rating={stats.averageRating} />
      <span>{stats.reviewCount} ratings</span>
    </div>
  );
}
```
