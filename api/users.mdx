## Overview

The Users API handles user profiles, onboarding, profile visibility settings, data export, and account deletion. All operations use Convex Auth for authentication.

## Queries

### me

Get the current authenticated user.

```typescript
export const me = query({
  args: {},
  handler: async (ctx) => {
    const userId = await getAuthUserId(ctx);
    if (!userId) return null;
    return await ctx.db.get(userId);
  },
});
```

**Returns:** Current user document or `null` if not authenticated.

## Mutations

### ensureProfile

Create or update user profile on login.

```typescript
export const ensureProfile = mutation({
  args: {
    username: v.optional(v.string()),
    displayName: v.optional(v.string()),
  },
  handler: async (ctx, args) => { ... },
});
```

<ParamField path="username" type="string">
  Desired username (sanitized to alphanumeric + underscore)
</ParamField>

<ParamField path="displayName" type="string">
  Display name for profile
</ParamField>

**Returns:** User ID

**Errors:**
- `"Not authenticated"` - No valid session
- `"Username already taken"` - Username conflict

### updateProfile

Update user profile information.

```typescript
export const updateProfile = mutation({
  args: {
    displayName: v.optional(v.string()),

    favoriteShowIds: v.optional(v.array(v.id("shows"))),
    favoriteGenres: v.optional(v.array(v.string())),
    profileVisibility: v.optional(ProfileVisibility),
  },
  handler: async (ctx, args) => { ... },

```

<ParamField path="displayName" type="string">
  New display name
</ParamField>

<ParamField path="bio" type="string">
  Profile biography
</ParamField>

<ParamField path="username" type="string">
  New username (must be unique)
</ParamField>

<ParamField path="avatarStorageId" type="Id<'_storage'>">
  Uploaded avatar file ID
</ParamField>

<ParamField path="favoriteShowIds" type="Id<'shows'>[]">
  Up to 4 favorite show IDs

<ParamField path="favoriteGenres" type="string[]">

<ParamField path="profileVisibility" type="ProfileVisibility">

### deleteAccount

Permanently delete user account and all associated data.

```typescript
export const deleteAccount = mutation({
  args: {},
  handler: async (ctx) => { ... },
});
```

**Returns:** `{ success: true }`

**Deletes:**

## Profile Visibility

Profile visibility settings control who can see different sections:

```typescript
type ProfileVisibility = {
  favorites: "public" | "followers" | "mutuals" | "private";
  watchlist: "public" | "followers" | "mutuals" | "private";
  currentlyWatching: "public" | "followers" | "mutuals" | "private";
};
```

**Visibility levels:**
- `public` - Anyone can see
- `followers` - Only followers can see
- `mutuals` - Only mutual follows can see
- `private` - Only the user can see

**Default settings:**
```typescript
const DEFAULT_PROFILE_VISIBILITY = {
  favorites: "public",
  watchlist: "public",
  currentlyWatching: "followers",
};
```

## Example Usage

### Get Current User

```typescript
import { useQuery } from "convex/react";
import { api } from "../convex/_generated/api";

function ProfileButton() {
  const user = useQuery(api.users.me);
  
  if (!user) return <LoginButton />;
  
  return (
    <div>
      {user.displayName} (@{user.username})
    </div>
  );
}
```

### Update Profile

```typescript
import { useMutation } from "convex/react";
import { api } from "../convex/_generated/api";

function EditProfile() {
  const updateProfile = useMutation(api.users.updateProfile);
  
  const handleSubmit = async (bio: string) => {
    await updateProfile({ bio });
  };
  
  return <form onSubmit={handleSubmit}>...</form>;
}
```

### Search Users

```typescript
import { useQuery } from "convex/react";
import { api } from "../convex/_generated/api";

function UserSearch({ query }: { query: string }) {
  const results = useQuery(
    api.users.search,
    query.length >= 2 ? { text: query, limit: 10 } : "skip"
  );
  
  return (
    <div>
      {results?.map(user => (
        <UserCard key={user.user._id} preview={user} />
      ))}
    </div>
  );
}
```
